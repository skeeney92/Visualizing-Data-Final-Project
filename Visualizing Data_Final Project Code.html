fig, ax1 = plt.subplots(figsize=(10, 6))


COLOR_WC = 'blue'
COLOR_PED = 'gray'


sns.lineplot(
    data=df_wide,
    x='Y',
    y='WC',
    label='WC',  
    linewidth=2.5,
    color=COLOR_WC,
    ax=ax1        
)


ax1.set_ylabel('WC (Wolf Count)', fontsize=12, color=COLOR_WC)
ax1.set_xlabel('Year', fontsize=12)
ax1.set_ylim(0, 180) 
ax1.tick_params(axis='y', labelcolor=COLOR_WC)



ax2 = ax1.twinx() 



sns.lineplot(
    data=df_wide,
    x='Y',
    y='PED',
    label='PED', 
    linewidth=2.5,
    linestyle='--', 
    color=COLOR_PED,
    ax=ax2        
)


ax2.set_ylabel('PED', fontsize=12, color=COLOR_PED)

ax2.set_ylim(0, 20) 
ax2.tick_params(axis='y', labelcolor=COLOR_PED)





ax1.set_title('WC and PED by Year, 1995-2022', fontsize=14, pad=20)


h1, l1 = ax1.get_legend_handles_labels()
h2, l2 = ax2.get_legend_handles_labels()

ax1.legend(h1 + h2, l1 + l2, loc='upper left')


plt.figtext(0.5, 0.01,
            "Source: Cassidy et al., 2024",
            wrap=True,
            horizontalalignment='center',
            fontsize=9)


fig.tight_layout() 
plt.show()
